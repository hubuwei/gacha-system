### 3. 创建 HTTP 请求工具

在 `game-mall/src/utils/` 下创建 `request.js`:

```javascript
import { API_CONFIG } from '../config';

/**
 * HTTP 请求封装
 */
class Request {
  constructor() {
    this.baseURL = API_CONFIG.baseURL;
    this.timeout = API_CONFIG.timeout;
  }

  /**
   * 请求方法
   */
  async request(options) {
    const { url, method = 'GET', data, headers = {} } = options;

    // 添加 Token
    const token = localStorage.getItem('token');
    if (token) {
      headers['Authorization'] = `Bearer ${token}`;
    }

    try {
      const response = await fetch(`${this.baseURL}${url}`, {
        method,
        headers: {
          'Content-Type': 'application/json',
          ...headers,
        },
        body: method !== 'GET' ? JSON.stringify(data) : null,
      });

      const result = await response.json();

      if (result.code === 200) {
        return result.data;
      } else {
        throw new Error(result.message || '请求失败');
      }
    } catch (error) {
      console.error('请求错误:', error);
      throw error;
    }
  }

  // GET 请求
  get(url, params) {
    return this.request({ url, method: 'GET', data: params });
  }

  // POST 请求
  post(url, data) {
    return this.request({ url, method: 'POST', data });
  }

  // PUT 请求
  put(url, data) {
    return this.request({ url, method: 'PUT', data });
  }

  // DELETE 请求
  delete(url) {
    return this.request({ url, method: 'DELETE' });
  }
}

export const http = new Request();
export default http;
```

##  前端集成示例

### 1. 用户登录集成

修改 `game-mall/src/App.jsx`:

```javascript
import { useState } from 'react';
import { userApi } from './api/user';
import { gameApi } from './api/game';

function App() {
  const [user, setUser] = useState(null);
  const [games, setGames] = useState([]);

  // 登录处理
  const handleLogin = async (username, password) => {
    try {
      const result = await userApi.login({ username, password });
      setUser(result.user);
      localStorage.setItem('token', result.token);
      console.log('登录成功');
    } catch (error) {
      console.error('登录失败:', error);
      alert(error.message);
    }
  };

  // 获取游戏列表
  const loadGames = async () => {
    try {
      const data = await gameApi.getGames();
      setGames(data);
    } catch (error) {
      console.error('加载游戏失败:', error);
    }
  };

  // 登出
  const handleLogout = () => {
    localStorage.removeItem('token');
    setUser(null);
  };

  return (
    <div className="app">
      {/* 导航栏 */}
      <nav className="navbar">
        {user ? (
          <div className="user-info">
            <span>欢迎，{user.nickname || user.username}</span>
            <button onClick={handleLogout}>退出</button>
          </div>
        ) : (
          <button onClick={() => handleLogin('testuser', 'test123')}>
            登录
          </button>
        )}
      </nav>

      {/* 游戏列表 */}
      <div className="games-grid">
        {games.map(game => (
          <div key={game.id} className="game-card">
            <h3>{game.title}</h3>
            <p>¥{game.currentPrice}</p>
          </div>
        ))}
      </div>
    </div>
  );
}

export default App;
```

### 2. 购物车集成

```javascript
import { useEffect, useState } from 'react';
import { cartApi } from './api/cart';

function ShoppingCart() {
  const [cartItems, setCartItems] = useState([]);
  const [loading, setLoading] = useState(false);

  // 加载购物车
  useEffect(() => {
    loadCart();
  }, []);

  const loadCart = async () => {
    try {
      setLoading(true);
      const items = await cartApi.getCart();
      setCartItems(items);
    } catch (error) {
      console.error('加载购物车失败:', error);
    } finally {
      setLoading(false);
    }
  };

  // 添加到购物车
  const addToCart = async (gameId) => {
    try {
      await cartApi.addToCart(gameId, 1);
      await loadCart(); // 重新加载
      alert('已添加到购物车');
    } catch (error) {
      alert('添加失败');
    }
  };

  // 从购物车删除
  const removeFromCart = async (itemId) => {
    try {
      await cartApi.deleteCartItem(itemId);
      await loadCart();
    } catch (error) {
      console.error('删除失败:', error);
    }
  };

  return (
    <div className="shopping-cart">
      <h2>购物车</h2>
      {loading ? (
        <div>加载中...</div>
      ) : (
        <div className="cart-items">
          {cartItems.map(item => (
            <div key={item.id} className="cart-item">
              <span>{item.gameTitle}</span>
              <span>¥{item.actualPrice}</span>
              <button onClick={() => removeFromCart(item.id)}>
                删除
              </button>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}

export default ShoppingCart;
```

## 🐛 常见问题排查

### 1. CORS 错误
**现象**: Access to fetch at 'xxx' from origin 'null' has been blocked

**解决**: 
- 检查后端 SecurityConfig 是否配置 CORS
- 确保前端请求的 URL 正确

### 2. Token 失效
**现象**: 401 Unauthorized

**解决**:
- 检查 Token 是否过期
- 重新登录获取新 Token
- 检查请求头格式：Authorization: Bearer {token}

### 3. 数据格式不匹配
**现象**: 400 Bad Request

**解决**:
- 检查请求参数格式
- 确保字段名一致
- 检查数据类型（String/Number/Boolean）

### 4. 网络错误
**现象**: Failed to fetch

**解决**:
- 检查后端服务是否启动
- 检查端口是否正确
- 检查防火墙设置

```javascript
import useSWR from 'swr';

const fetcher = (url) => fetch(url).then((res) => res.json());

function Games() {
  const { data, error } = useSWR('/api/games', fetcher);
  
  if (error) return <div>加载失败</div>;
  if (!data) return <div>加载中...</div>;
  
  return <div>{/* 渲染数据 */}</div>;
}
```
